    <script>
        // 데이터 정의
        const themeData = {
            name: "중동전쟁",
            children: [
                { name: "방산", color: "#ef4444", children: [{name: "LIG넥스원"}, {name: "한화에어로"}, {name: "풍산"}] },
                { name: "석유", color: "#f59e0b", children: [{name: "흥구석유"}, {name: "중앙에너비스"}, {name: "한국석유"}] },
                { name: "재건", color: "#10b981", children: [{name: "전진건설"}, {name: "SG"}, {name: "남광토건"}] },
                { name: "신재생", color: "#3b82f6", children: [{name: "한화솔루션"}, {name: "OCI"}, {name: "씨에스윈드"}] },
                { name: "식품", color: "#8b5cf6", children: [{name: "한일사료"}, {name: "팜스토리"}, {name: "미래생명"}] },
                { name: "해운", color: "#06b6d4", children: [{name: "HMM"}, {name: "팬오션"}] },
                { name: "보안", color: "#ec4899", children: [{name: "안랩"}, {name: "윈스"}] },
                { name: "원자재", color: "#eab308", children: [{name: "엘켐텍"}, {name: "포스코엠텍"}] }
            ]
        };

        const width = document.getElementById('viz').clientWidth;
        const height = 600;
        const tooltip = d3.select("#tooltip");

        function clearViz() {
            d3.select("#viz").selectAll("*").remove();
            d3.selectAll(".tab-btn").classed("active", false);
        }

        // --- 1. Sankey Diagram ---
        function renderSankey() {
            clearViz();
            d3.select("#btn-sankey").classed("active", true);
            const svg = d3.select("#viz").append("svg").attr("width", width).attr("height", height).append("g").attr("transform", "translate(50,50)");

            const sankey = d3.sankey()
                .nodeWidth(15)
                .nodePadding(10)
                .extent([[0, 0], [width - 100, height - 100]]);

            let nodes = [{ name: "중동전쟁" }];
            let links = [];

            themeData.children.forEach((theme, i) => {
                nodes.push({ name: theme.name, color: theme.color });
                links.push({ source: 0, target: nodes.length - 1, value: theme.children.length });
                let themeIdx = nodes.length - 1;
                theme.children.forEach(stock => {
                    nodes.push({ name: stock.name, color: theme.color });
                    links.push({ source: themeIdx, target: nodes.length - 1, value: 1 });
                });
            });

            const { nodes: skNodes, links: skLinks } = sankey({
                nodes: nodes.map(d => Object.assign({}, d)),
                links: links.map(d => Object.assign({}, d))
            });

            svg.append("g")
                .selectAll("rect")
                .data(skNodes)
                .join("rect")
                .attr("x", d => d.x0)
                .attr("y", d => d.y0)
                .attr("height", d => d.y1 - d.y0)
                .attr("width", d => d.x1 - d.x0)
                .attr("fill", d => d.color || "#64748b")
                .attr("rx", 2)
                .on("mouseover", (e, d) => {
                    tooltip.style("opacity", 1).html(`<strong>${d.name}</strong>`);
                })
                .on("mouseout", () => tooltip.style("opacity", 0));

            svg.append("g")
                .selectAll("path")
                .data(skLinks)
                .join("path")
                .attr("class", "sankey-link")
                .attr("d", d3.sankeyLinkHorizontal())
                .attr("stroke", d => d.source.color || "#cbd5e1")
                .attr("stroke-width", d => Math.max(1, d.width));

            svg.append("g")
                .selectAll("text")
                .data(skNodes)
                .join("text")
                .attr("x", d => d.x0 < width / 2 ? d.x1 + 6 : d.x0 - 6)
                .attr("y", d => (d.y1 + d.y0) / 2)
                .attr("dy", "0.35em")
                .attr("text-anchor", d => d.x0 < width / 2 ? "start" : "end")
                .attr("class", "node-label")
                .text(d => d.name);
        }

        // --- 2. Sunburst Chart ---
        function renderSunburst() {
            clearViz();
            d3.select("#btn-sunburst").classed("active", true);
            const radius = Math.min(width, height) / 2;
            const svg = d3.select("#viz").append("svg")
                .attr("width", width).attr("height", height)
                .append("g").attr("transform", `translate(${width/2},${height/2})`);

            const root = d3.hierarchy(themeData)
                .sum(d => 1)
                .sort((a, b) => b.value - a.value);

            const partition = d3.partition().size([2 * Math.PI, radius]);
            partition(root);

            const arc = d3.arc()
                .startAngle(d => d.x0)
                .endAngle(d => d.x1)
                .innerRadius(d => d.y0)
                .outerRadius(d => d.y1);

            svg.selectAll("path")
                .data(root.descendants())
                .join("path")
                .attr("d", arc)
                .attr("fill", d => {
                    if (d.depth === 0) return "#f1f5f9";
                    if (d.depth === 1) return d.data.color;
                    return d.parent.data.color;
                })
                .attr("fill-opacity", d => d.depth === 2 ? 0.6 : 1)
                .attr("stroke", "white")
                .on("mouseover", (e, d) => {
                    tooltip.style("opacity", 1).html(`<strong>${d.data.name}</strong>`);
                })
                .on("mouseout", () => tooltip.style("opacity", 0));

            svg.selectAll("text")
                .data(root.descendants().filter(d => d.depth > 0 && (d.x1 - d.x0) > 0.1))
                .join("text")
                .attr("transform", function(d) {
                    const x = (d.x0 + d.x1) / 2 * 180 / Math.PI;
                    const y = (d.y0 + d.y1) / 2;
                    return `rotate(${x - 90}) translate(${y},0) rotate(${x < 180 ? 0 : 180})`;
                })
                .attr("dy", "0.35em")
                .attr("text-anchor", "middle")
                .style("font-size", "9px")
                .text(d => d.data.name);
        }

        // --- 3. Chord Diagram (Simplified) ---
        function renderChord() {
            clearViz();
            d3.select("#btn-chord").classed("active", true);
            const outerRadius = Math.min(width, height) * 0.5 - 40;
            const innerRadius = outerRadius - 30;

            const svg = d3.select("#viz").append("svg")
                .attr("width", width).attr("height", height)
                .append("g").attr("transform", `translate(${width/2},${height/2})`);

            // 테마 이름 추출
            const names = themeData.children.map(d => d.name);
            const matrix = Array.from({length: names.length}, () => Array(names.length).fill(0));
            
            // 학습용 임의 연결 데이터 (테마간의 상관관계 가정)
            themeData.children.forEach((d, i) => {
                matrix[i][i] = 5; // 자기 자신 비중
                if(i < names.length - 1) matrix[i][i+1] = 2; // 인접 테마 연관성
                if(i > 0) matrix[i][i-1] = 2;
            });

            const chord = d3.chord().padAngle(0.05).sortSubgroups(d3.descending)(matrix);

            const arc = d3.arc().innerRadius(innerRadius).outerRadius(outerRadius);
            const ribbon = d3.ribbon().radius(innerRadius);

            const group = svg.append("g")
                .selectAll("g")
                .data(chord.groups)
                .join("g");

            group.append("path")
                .attr("fill", d => themeData.children[d.index].color)
                .attr("d", arc);

            group.append("text")
                .each(d => { d.angle = (d.startAngle + d.endAngle) / 2; })
                .attr("dy", "0.35em")
                .attr("transform", d => `
                    rotate(${(d.angle * 180 / Math.PI - 90)})
                    translate(${outerRadius + 10})
                    ${d.angle > Math.PI ? "rotate(180)" : ""}
                `)
                .attr("text-anchor", d => d.angle > Math.PI ? "end" : "start")
                .style("font-size", "12px")
                .text(d => names[d.index]);

            svg.append("g")
                .attr("fill-opacity", 0.67)
                .selectAll("path")
                .data(chord)
                .join("path")
                .attr("d", ribbon)
                .attr("fill", d => themeData.children[d.source.index].color)
                .attr("stroke", d => d3.rgb(themeData.children[d.source.index].color).darker());
        }

        function switchChart(type) {
            if (type === 'sankey') renderSankey();
            else if (type === 'sunburst') renderSunburst();
            else if (type === 'chord') renderChord();
        }

        window.onload = renderSankey;
        window.onresize = () => location.reload();
    </script>
