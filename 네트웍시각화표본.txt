        const svg = d3.select("#viz").append("svg")
            .attr("width", width).attr("height", height);

        // 물리 엔진 설정
        const simulation = d3.forceSimulation(data.nodes)
            .force("link", d3.forceLink(data.links).id(d => d.id).distance(120))
            .force("charge", d3.forceManyBody().strength(-450)) // 반발력 강화
            .force("center", d3.forceCenter(width / 2, height / 2))
            .force("collision", d3.forceCollide().radius(d => d.r + 10));

        // 연결선
        const link = svg.append("g")
            .selectAll("line")
            .data(data.links)
            .join("line")
            .attr("class", "link")
            .attr("stroke-width", d => d.value / 2.5);

        // 노드 그룹
        const node = svg.append("g")
            .selectAll("g")
            .data(data.nodes)
            .join("g")
            .attr("class", "node")
            .call(d3.drag()
                .on("start", dragstarted)
                .on("drag", dragged)
                .on("end", dragended));

        // 노드 원형
        node.append("circle")
            .attr("r", d => d.r)
            .attr("fill", d => d.color)
            .attr("stroke", d => d.type === 'theme' ? "#000" : "#fff")
            .attr("stroke-width", d => d.type === 'theme' ? 1.5 : 2)
            .on("mouseover", (e, d) => {
                d3.select("#tooltip").style("opacity", 1).html(`
                    <div class="font-bold text-lg">${d.id}</div>
                    <div class="text-xs text-slate-500">${d.type === 'theme' ? '섹터 테마' : '개별 종목'}</div>
                `);
            })
            .on("mousemove", (e) => {
                d3.select("#tooltip").style("left", (e.pageX + 15) + "px").style("top", (e.pageY - 15) + "px");
            })
            .on("mouseout", () => d3.select("#tooltip").style("opacity", 0));

        // 텍스트 라벨
        node.append("text")
            .attr("class", d => d.type === 'theme' ? "theme-label" : "label")
            .attr("dy", d => d.type === 'theme' ? 4 : d.r + 14)
            .attr("text-anchor", "middle")
            .text(d => d.id);

        // 틱(Tick) 애니메이션
        simulation.on("tick", () => {
            link
                .attr("x1", d => d.source.x)
                .attr("y1", d => d.source.y)
                .attr("x2", d => d.target.x)
                .attr("y2", d => d.target.y);

            node
                .attr("transform", d => `translate(${d.x},${d.y})`);
        });

        // 드래그 제어 함수
        function dragstarted(event) {
            if (!event.active) simulation.alphaTarget(0.3).restart();
            event.subject.fx = event.subject.x;
            event.subject.fy = event.subject.y;
        }
        function dragged(event) {
            event.subject.fx = event.x;
            event.subject.fy = event.y;
        }
        function dragended(event) {
            if (!event.active) simulation.alphaTarget(0);
            event.subject.fx = null;
            event.subject.fy = null;
        }

        // 반응형 처리
        window.addEventListener("resize", () => {
            const newWidth = document.getElementById('viz').clientWidth;
            svg.attr("width", newWidth);
            simulation.force("center", d3.forceCenter(newWidth / 2, height / 2)).alpha(0.3).restart();
        });
